Before you start, connect your free Supabase database.
This takes about 3 minutes.

Quick setup:
1. Go to supabase.com → create a free account
2. Create a new project (any name, any region)
3. Go to Project Settings → API
4. Copy your Project URL and anon public key
5. Paste both below, then run the SQL setup
Also run this SQL once in Supabase → SQL Editor:

create table if not exists users ( id uuid primary key default gen_random_uuid(), username text unique not null, pin text not null, avatar text default '🙂', color text default '#c84b31', created_at timestamptz default now() ); create table if not exists todos ( id uuid primary key default gen_random_uuid(), user_id uuid references users(id) on delete cascade, text text not null, cat text, due date, done boolean default false, is_focus boolean default false, created_at timestamptz default now() ); create table if not exists meetings ( id uuid primary key default gen_random_uuid(), user_id uuid references users(id) on delete cascade, time text not null, title text not null, created_at timestamptz default now() ); create table if not exists journal ( id uuid primary key default gen_random_uuid(), user_id uuid references users(id) on delete cascade, entry_key text not null, body text not null, created_at timestamptz default now(), unique(user_id, entry_key) );
